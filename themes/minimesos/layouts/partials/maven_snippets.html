<repository>
  <id>jitpack.io</id>
  <url>https://jitpack.io</url>
</repository>

Step 2. Add the dependency in the form:

<dependency>
  <groupId>com.github.ContainerSolutions</groupId>
  <artifactId>minimesos</artifactId>
  <version>{{ index $latestRelease "tag_name" }}</version>
</dependency>

Step 3. Add a @ClassRule in your tests to start and destroy your Mesos test cluster

@ClassRule
public static final MesosClusterTestRule RULE =
  MesosClusterTestRule.fromFile("src/test/resources/discoverySystemTest");

public static final MesosCluster CLUSTER = RULE.getMesosCluster();
